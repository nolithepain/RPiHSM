79      /**
80       * Generates all the common elements of the stages.
81       *
82       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
83       * @param userPath     user home directory on the Raspberry Pi
84       */
85      public AbstractStage(SerialHelperI serialHelper, String userPath) {
86          super();
87  
88          this.userPath = userPath;
89          this.serialHelper = serialHelper;
90          this.b = ResourceBundle.getBundle("language", Locale.getDefault(), new UTF8Control());
91          this.setTitle(TITLE); //Application title
92  
93          //Creates the Label for error and success messages
94          messages = new Label();
95          messages.setTextAlignment(TextAlignment.CENTER);
96          messages.setPadding(new Insets(10));
97          messages.prefWidth(WIDTH);
98  
99          //Creates the GridPane to place the objects on the stage
100         grid = new GridPane();
101         grid.setAlignment(Pos.CENTER);
102         grid.setHgap(PADDING_SMALL);
103         grid.setVgap(PADDING_SMALL);
104         grid.setPadding(new Insets(PADDING_MEDIUM));
105 
106 
107         //Creates the VBox
108         pane = new VBox();
109         pane.setAlignment(Pos.TOP_CENTER);
110         pane.getChildren().add(this.messages);
111         pane.getChildren().add(grid);
112         pane.setStyle(PANE_BACKGROUND);
113         grid.setPrefWidth(WIDTH);
114 
115         //sets and shows the scene
116         this.setScene(new Scene(pane, WIDTH, HEIGHT));
117         this.show();
118 
119         this.setOnCloseRequest(e -> {
120             if (serialHelper != null) {
121                 try {
122                     serialHelper.closeConnection(); //closes the serial connection while closing the application
123                 } catch (SerialPortException e1) {
124                     error(b.getString("error.com.unsupported.operation"));
125                 }
126             }
127         });
128     }

131     /**
132      * Generates the header elements of the application.
133      *
134      * @param backToMenu true if a button to go back in the menu must be displayed otherwise false
135      */
136     protected void addHeader(boolean backToMenu) {
137         sceneTitle.setFont(Font.font(FONT, FontWeight.NORMAL, TITLE_FONT_SIZE));
138         sceneTitle.setTextFill(Paint.valueOf(HEADER_TITLE_COLOR));
139         ImageView logo = new ImageView(this.getClass().getResource(HEADER_IMAGE_PATH).toExternalForm());
140 
141         //Creates the GridPane for the header structure
142         GridPane header = new GridPane();
143         header.setMinSize(WIDTH, HEADER_HEIGHT);
144         header.setPadding(new Insets(15, 12, 15, 12));
145         header.setStyle(HEADER_BACKGROUND);
146         ColumnConstraints col1 = new ColumnConstraints();
147         ColumnConstraints col2 = new ColumnConstraints();
148         ColumnConstraints col3 = new ColumnConstraints();
149         col1.setPercentWidth(25);
150         col2.setPercentWidth(50);
151         col3.setPercentWidth(23);
152 
153         header.add(logo, 0, 0);
154         header.add(sceneTitle, 1, 0);
155         header.getColumnConstraints().addAll(col1, col2, col3);
156 
157         GridPane.setHalignment(logo, HPos.LEFT);
158         GridPane.setHalignment(sceneTitle, HPos.CENTER);
159 
160         ComboBox<String> languageComboBox = new ComboBox<>();
161         languageComboBox.getItems().addAll(b.getString("constants.languages").split(","));
162         languageComboBox.getSelectionModel().select(Locale.getDefault().getLanguage());
163         languageComboBox.setStyle(LANGUAGE_BUTTON_STYLE);
164         languageComboBox.setPrefSize(LANGUAGE_BUTTON_WIDTH, BACK_BUTTON_HEIGHT);
165 
166         languageComboBox.valueProperty().addListener((obs, oldValue, newValue) -> {
167         	System.setProperty("user.language", newValue);
168             Locale.setDefault(Locale.forLanguageTag(newValue));
169         });
170 
171         if (backToMenu) { //Creates the backToMenu button if needed
172             Button menuButton = new Button(b.getString("button.text.back"));
173             menuButton.setPrefSize(BACK_BUTTON_WIDTH, BACK_BUTTON_HEIGHT);
174             menuButton.setOnMouseClicked(e -> {
175                 this.hide();
176                 new MainStage(serialHelper, userPath);
177             });
178             menuButton.setStyle(BUTTONS_STYLE);
179             header.add(new HBox(menuButton, languageComboBox), 2, 0);
180             GridPane.setHalignment(menuButton, HPos.RIGHT);
181         } else {
182             header.add(languageComboBox, 2, 0);
183         }
184 
185         pane.getChildren().add(0, header); //adds the header on first position (on top)
186     }

205         messages.setText(message); //prints the success message in green
206     }

208     /**
209      * Checks if a given key set exist using the {@link KeyExistence} class.
210      *
211      * @param keySet name to check
212      */
213     protected void checkKeyExistence(String keySet) {
214         KeyExistence ke = new KeyExistence(serialHelper, userPath, keySet);
215         try {
216             if (ke.keyexist()) success(b.getString("command.success.keyset.exist")); //checks if the key set exists
217             else error(b.getString("command.error.keyset.not.exist"));
218         } catch (OperationNotSupportedException e) {
219             error("error.unsupported.operation");
220         }
221     }
