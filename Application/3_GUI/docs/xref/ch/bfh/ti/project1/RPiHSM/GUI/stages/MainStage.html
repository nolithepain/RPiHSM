10  /**
11   * <h1>MainStageMainStage</h1>
12   * An implementation of {@link AbstractStage}. This is the main stage of the application where the are the link with all the supported operations.
13   */
14  public class MainStage extends AbstractStage {
15  
16      private Button addKeyButton, createKeySetButton, deleteKeySetButton, decryptButton, encryptButton, promoteDemoteButton, pubKeyButton, revokeButton, signButton, verifyButton;
17  
18      /**
19       * Generates all the graphical elements for the supported operations.
20       *
21       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
22       * @param userPath     user home directory on the Raspberry Pi
23       */
24      public MainStage(SerialHelperI serialHelper, String userPath) {
25          super(serialHelper, userPath);
26  
27          //creates the scene objects
28          this.sceneTitle = new Label(b.getString("stage.title.main"));
29          addKeyButton = new Button(b.getString("button.text.main.stage.add.key"));
30          createKeySetButton = new Button(b.getString("button.text.main.stage.create.keyset"));
31          deleteKeySetButton = new Button(b.getString("button.text.main.stage.delete.keyset"));
32          decryptButton = new Button(b.getString("button.text.main.stage.decrypt"));
33          encryptButton = new Button(b.getString("button.text.main.stage.encrypt"));
34          promoteDemoteButton = new Button(b.getString("button.text.main.stage.promote.demote"));
35          pubKeyButton = new Button(b.getString("button.text.main.stage.public.key"));
36          revokeButton = new Button(b.getString("button.text.main.stage.revoke"));
37          signButton = new Button(b.getString("button.text.main.stage.sign"));
38          verifyButton = new Button(b.getString("button.text.main.stage.verify"));
39  
40          //creates container titles
41          Label titleContainer1 = new Label(b.getString("stage.containers.title.keyset"));
42          Label titleContainer2 = new Label(b.getString("stage.containers.title.key"));
43          Label titleContainer3 = new Label(b.getString("stage.containers.title.crypto"));
44          Label titleContainer4 = new Label(b.getString("stage.containers.title.signature"));
45  
46          titleContainer1.setStyle(SECTIONS_TITLES_STYLE);
47          titleContainer2.setStyle(SECTIONS_TITLES_STYLE);
48          titleContainer3.setStyle(SECTIONS_TITLES_STYLE);
49          titleContainer4.setStyle(SECTIONS_TITLES_STYLE);
50  
51          //creates subcontainers
52          HBox subContainer1 = new HBox();
53          HBox subContainer2 = new HBox();
54          HBox subContainer3 = new HBox();
55          HBox subContainer4 = new HBox();
56  
57          //Places the buttons in the right sub container
58          subContainer1.getChildren().addAll(createKeySetButton, deleteKeySetButton, pubKeyButton);
59          subContainer2.getChildren().addAll(addKeyButton, promoteDemoteButton, revokeButton);
60          subContainer3.getChildren().addAll(encryptButton, decryptButton);
61          subContainer4.getChildren().addAll(signButton, verifyButton);
62  
63          //creates the 4 sections
64          VBox container1 = new VBox(titleContainer1, subContainer1);
65          VBox container2 = new VBox(titleContainer2, subContainer2);
66          VBox container3 = new VBox(titleContainer3, subContainer3);
67          VBox container4 = new VBox(titleContainer4, subContainer4);
68  
69          //Creates the main container
70          VBox masterContainer = new VBox(container1, container2, container3, container4);
71  
72          //Sets the paddings
73          masterContainer.setPadding(new Insets(PADDING_SMALL));
74          container1.setPadding(new Insets(PADDING_SMALL));
75          container2.setPadding(new Insets(PADDING_SMALL));
76          container3.setPadding(new Insets(PADDING_SMALL));
77          container4.setPadding(new Insets(PADDING_SMALL));
78  
79          //Sets the vertical spacing
80          masterContainer.setSpacing(PADDING_SMALL);
81          container1.setSpacing(10);
82          container2.setSpacing(10);
83          container3.setSpacing(10);
84          container4.setSpacing(10);
85  
86          //sets the horizontal spacing
87          subContainer1.setSpacing(PADDING_SMALL);
88          subContainer2.setSpacing(PADDING_SMALL);
89          subContainer3.setSpacing(PADDING_SMALL);
90          subContainer4.setSpacing(PADDING_SMALL);
91  
92          //Sets the buttons size
93          addKeyButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
94          createKeySetButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
95          deleteKeySetButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
96          decryptButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
97          encryptButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
98          promoteDemoteButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
99          pubKeyButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
100         revokeButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
101         signButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
102         verifyButton.setMinWidth(MAIN_MENU_BUTTON_SIZE);
103 
104         //Sets the buttons style
105         addKeyButton.setStyle(BUTTONS_STYLE);
106         createKeySetButton.setStyle(BUTTONS_STYLE);
107         deleteKeySetButton.setStyle(BUTTONS_STYLE);
108         decryptButton.setStyle(BUTTONS_STYLE);
109         encryptButton.setStyle(BUTTONS_STYLE);
110         promoteDemoteButton.setStyle(BUTTONS_STYLE);
111         pubKeyButton.setStyle(BUTTONS_STYLE);
112         revokeButton.setStyle(BUTTONS_STYLE);
113         signButton.setStyle(BUTTONS_STYLE);
114         verifyButton.setStyle(BUTTONS_STYLE);
115 
116         //Sets the containers style
117         container1.setStyle(CONTAINERS_STYLE);
118         container2.setStyle(CONTAINERS_STYLE);
119         container3.setStyle(CONTAINERS_STYLE);
120         container4.setStyle(CONTAINERS_STYLE);
121 
122         //Sets the actions for each button
123         //each button opens a new stage
124         addKeyButton.setOnMouseClicked(e -> {
125             this.hide();
126             new AddKeyCommandStage(serialHelper, userPath);
127         });
128         createKeySetButton.setOnMouseClicked(e -> {
129             this.hide();
130             new CreateKeySetCommandStage(serialHelper, userPath);
131         });
132         deleteKeySetButton.setOnMouseClicked(e -> {
133             this.hide();
134             new DeleteKeySetCommandStage(serialHelper, userPath);
135         });
136         encryptButton.setOnMouseClicked(e -> {
137             this.hide();
138             new EncryptDecryptCommandStage(serialHelper, userPath, true);
139         });
140         decryptButton.setOnMouseClicked(e -> {
141             this.hide();
142             new EncryptDecryptCommandStage(serialHelper, userPath, false);
143         });
144         promoteDemoteButton.setOnMouseClicked(e -> {
145             this.hide();
146             new PromoteDemoteCommandStage(serialHelper, userPath);
147         });
148         pubKeyButton.setOnMouseClicked(e -> {
149             this.hide();
150             new PublicKeyCommandStage(serialHelper, userPath);
151         });
152         revokeButton.setOnMouseClicked(e -> {
153             this.hide();
154             new RevokeCommandStage(serialHelper, userPath);
155         });
156         signButton.setOnMouseClicked(e -> {
157             this.hide();
158             new SignCommandStage(serialHelper, userPath);
159         });
160         verifyButton.setOnMouseClicked(e -> {
161             this.hide();
162             new VerifyCommandStage(serialHelper, userPath);
163         });
164 
165         //creates the header and adds the master container to the stage grid
166         addHeader(false);
167         grid.add(masterContainer, 0, 0);
168     }
